CREATE TABLE emails (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    sender TEXT NOT NULL,
    real_sender TEXT NOT NULL,
    receiver TEXT NOT NULL,
    subject TEXT,
    body TEXT,
    received_at TEXT NOT NULL
);
CREATE TABLE sends (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    sender TEXT NOT NULL,
    receiver TEXT NOT NULL,
    subject TEXT,
    body TEXT,
    sent_at TEXT NOT NULL
);
